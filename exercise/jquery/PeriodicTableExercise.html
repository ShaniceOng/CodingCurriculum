Dr Java has just shown chemist Antoine Lavoisier our Periodic Table of Elements and he was very impressed! He is exploring new ways to store the data of each element and Dr Java thinks you could help him by using classes and objects (and a few other tricks) to create a Chemical Elements Record in Java!
Make a program that would allow Lavoisier to enter each element's name, mass number and category (metal, metalloid, non-metal). This program should also update the atomic number of each element whenever another element is added to the record.


ChemicalElement

This class has the attributes of an element and setter and getter methods to give other classes access to this information. Use constructor overloading to allow the user to put an element into the record without knowing its category.

ChemicalElementsExercise

This class will hold the main method and carry out the input and output methods. It calls the methods of the other classes and determines the flow between them.

ChemicalElementRecord

The record class stores the current collection of elements and handles the addition and removal of elements.

ElementProperty

ElementProperty has instance variables for Chemical Properties and a method to print these attributes. This class is created to be inherited by the property classes.

ElementSorter

Dr Java wants this program to sort the elements according to mass number then assign atomic numbers. This would update the list every time a newly-discovered element is added. Use this a method to sort the elements according to mass number then assign atomic numbers according to this arrangement.

MetalloidProperty

This class inherits the ElementProperty class and has attributes specific to metalloids.

MetalProperty

This class inherits the ElementProperty class and has attributes specific to metals.

NonMetalProperty

This class inherits the ElementProperty class and has attributes specific to non-metals.

      if (category.compareToIgnoreCase("metal")==0)
      {
          this.category = new MetalProperty();

      atomicNumber = atomNum;

      return massNumber;
  }

  }

        c = (int) length;
    }
  }

import java.util.Scanner;

public class ShapesExercise {
  private String getStringInput (String question)
  {
    System.out.println(question);

    return scan.nextLine();
  }

  private int getIntInput(String question)
  {
    System.out.println(question);
    Scanner in = new Scanner(System.in);
    return in.nextInt();
  }

  private int shapeMainMenu ()
  {
    System.out.println("1. Create a shape");
    System.out.println("2. Create and scale a shape");
    System.out.println("3. Quit");
    return getIntInput("Enter Option:");
  }

  private void userOption (int n)
  {
    switch (n)
    {
        case 1:
            createShape(shapeMenu()).printInformation();
            break;
        case 2:
            createAndScale(shapeMenu()).printInformation();
            break;
        default:
            break;
    }
  }
  private int shapeMenu ()
  {
    System.out.print("Choose a shape: ");
    System.out.println("1. Square");
    System.out.println("2. Triangle");
    System.out.println("3. Circle");
    int choice = getIntInput("Enter Option:");
    return choice;
  }

  private Shape createShape(int choice)
  {
    if (choice == 1)
    {
        int x = getIntInput("Square length:");
        int y = getIntInput("Square width:");
        Square square = new Square(x,y);
        return square;
    }
    else if (choice==2)
    {
        System.out.println("Two known lengths or three known lengths?");
        int input = getIntInput("Enter 2 or 3:");
        if (input==2) {
            int a = getIntInput("Length a:");
            int b = getIntInput("Length b:");
            int angle = getIntInput("Angle between a and b:");
            Triangle triangle = new Triangle(a,b,angle);
            return triangle;
        }
        else if (input == 3)
        {
            int a = getIntInput("Length a:");
            int b = getIntInput("Length b:");
            int c = getIntInput("Length c:");
            int angle = getIntInput("Angle opposite c:");
            Triangle triangle2 = new Triangle(a,b,c,angle);
            return triangle2;
        }
    }

    int radius = getIntInput("Radius:");
    Circle circle = new Circle(radius);
    return circle;
  }

  private Shape createAndScale (int choice)
  {
    Shape shape = createShape(choice);
    int scale = getIntInput("Enter scale factor: ");
    shape.scaleShape(scale);
    return shape;
  }

  public static void main (String[] args)
  {
    ShapesExercise shapes = new ShapesExercise();
    shapes.userOption(shapes.shapeMainMenu());
  }
