NumbersRange

  • A constructor: NumbersRange(int lower, int upper)
  • A getLower method to get the lower limit
  • A getUpper method to get the upper limit
  • A method that returns an array list of values in the range, in ascending order, inclusive of the upper and lower limits: ArrayList<Integer> getValues()
  • A method that checks if a number is in the range: boolean contains(int n)
  • A method to get the sum of all the numbers in the range: int sumRange()

NumbersRange


import java.util.ArrayList;
import java.util.ListIterator;

public class NumbersRange {
  private int lower;
  private int upper;
  private int difference;

  public NumbersRange (int lower, int upper)
  {
    this.lower = lower;
    this.upper = upper;
    difference = upper - lower;

  }

  private int getLower()
  {
    return this.lower;
  }

  private int getUpper()
  {
    return this.upper;
  }

  private ArrayList<Integer> getValues()
  {
    ArrayList<Integer> range = new ArrayList<>();
    for (int i=this.getLower(); i<=this.getUpper(); i++) {
        range.add(i);
    }
    return range;
  }

  public boolean contains (int n)
  {
    for (ListIterator <Integer> iter = this.getValues().listIterator(); iter.hasNext();)
    {
        int currentNum = iter.next();
        if (n==currentNum)
        {
            return true;
        }
    }
    return false;
  }

  public int sumRange ()
  {
    int sum = 0;

    for (int i : getValues())
    {
        sum += i;
    }
    return sum;
  }
}

public class Triangle implements Shape{
  int a;
  int b;
  int c;
  double angle;
  int area;

  public Triangle(int a, int b, int c, double angle)
  {
    this.a = a;
    this.b = b;
    this.c = c;
    this.angle = angle;
    findArea();
  }

  public Triangle(int a, int b, double angle)
  {
    this.a = a;
    this.b = b;
    this.angle = angle;
    pythagorasTheorem();
    findArea();
  }

  public void drawShape ()
  {
    System.out.println("A triangle with sides of lengths " + a + ", " + b + " and " + c);
  }

  public void scaleShape(int scale)
  {
    if (scale>0)
    {
        a = a*scale;
        b = b*scale;
        c = c*scale;
    }
    else
    {
        System.out.println("invalid scale option");
    }
  }

  public void pythagorasTheorem ()
  {
    if (this.angle == 90)
    {
        double sum = Math.pow(a, 2.0) + Math.pow(b, 2.0);
        c = (int) Math.sqrt(sum);
    }

    else
    {
        double length = Math.sqrt(Math.pow(a,2.0) + Math.pow(b, 2.0) + (a*b*Math.cos(Math.toRadians(angle))));
        c = (int) length;
    }
  }

  private void findArea()
  {
    double length = (double) c;
    double area = 0.5*(a*b*Math.sin(Math.toRadians(angle)));
    this.area = (int) area;

  }

  public boolean checkShape()
  {
    if ((a+b)>c && (b+c)>a && (a+c)>b)
    {
        return true;
    }
    return false;
  }

  public void printInformation()
  {
    System.out.println("Shape: Triangle");
    System.out.println("Lengths = " + a + ", " + b + " and " + c);

import java.util.ArrayList;
import java.util.ListIterator;

public class ElementSorter {
  private ArrayList <ChemicalElement> listOfElements;

  public ElementSorter (ArrayList <ChemicalElement> inputList)
  {
      listOfElements = new ArrayList<>();
      this.listOfElements = inputList;
      sortElements();
  }

  public void sortElements ()
  {
      for (int i=1; i <listOfElements.size(); i++)
      {
          ChemicalElement tempElement = listOfElements.get(i);
          int massNumTemp = tempElement.getMassNumber();
          int j = i-1;
          while (massNumTemp <listOfElements.get(j).getMassNumber() && j >=0)
          {
              listOfElements.set(j+1, listOfElements.get(j));
              j--;
          }
          listOfElements.set(j+1, tempElement);

      }

      int atomNum = 1;

      for (ChemicalElement element: listOfElements)
      {
          element.setAtomicNumber(atomNum);
          atomNum++;
      }
}

public ArrayList <ChemicalElement< getSortedElements ()
{
    return listOfElements;
}
